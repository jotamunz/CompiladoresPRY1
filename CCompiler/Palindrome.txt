
int main() {
    int n, reversedN, remainder, originalN, n;
    int remainder;
    read(n);
    originalN = n;
    reversedN = 0;

    // reversed integer is stored in reversedN
    while (n != 0) {
        remainder = n % 10;
        reversedN = reversedN * 10 + remainder;
        n /= 10;
    }

    // palindrome if orignalN and reversedN are equal
    if (originalN == reversedN){
        write('Y');
    }   
    else {
        write('N');
    }

    return 0;
}
